class RiskManager:
    """High‑level container for all portfolio‑level risk controls.

    The class encapsulates three complementary mechanisms that together form a
    layered defence against catastrophic loss:

    * **Dynamic position sizing** – Optimal fraction of capital per trade is
      computed via the Kelly criterion (see :meth:`calculate_kelly_fraction`).
    * **Empirical calibration** – Historical trade outcomes are binned by the
      model's prediction confidence so that payoff ratios (*R/L*) can evolve in
      step with model drift (see :meth:`update_historical_data`).
    * **Portfolio draw‑down guardrail** – A light‑weight monitor capable of
      halting the entire strategy when losses breach a user‑defined cap (see
      :func:`check_drawdown`).

    The Kelly sizing formula maximises the expected log‑utility of wealth
    growth, originally derived by Kelly_ (1956) and championed for financial
    markets by Thorp_ (2006).  In the general unequal‑payoff case the closed
    form is:

    .. math::

        f^* = \\operatorname*{arg\\,max}_{0 \\le f \\le 1} \\; [\\,p\\,\\ln(1+fR)
            + (1-p)\\,\\ln(1-fL)\\,],

    where *R* and *L* are the average fractional gain and loss, respectively.
    Because *R* and *L* are not stationary, the implementation solves the
    optimisation numerically each time using :pyfunc:`scipy.optimize.minimize_scalar`.

    Attributes
    ----------
    :ivar p_bins: list[float]
        Right‑closed edges of confidence intervals used to discretise model
        probabilities. Defaults to ``[0.5, 0.6, 0.7, 0.8, 0.9, 1.0]``. :noindex:

    :ivar R_avg: dict[float, list[float]]
        Running lists of realised trade pay‑offs keyed by :py:attr:`p_bins`.
        ``R_avg[bin]`` stores *positive* returns. :noindex:

    :ivar L_avg: dict[float, list[float]]
        Running lists of realised trade pay‑offs keyed by :py:attr:`p_bins`.
        ``L_avg[bin]`` stores *positive* absolute losses. :noindex:


    Examples
    --------
    >>> rm = RiskManager()
    >>> rm.update_historical_data(p=0.61, r=0.034)  # +3.4 % winner
    >>> rm.update_historical_data(p=0.58, r=-0.012) # −1.2 % loser
    >>> rm.calculate_kelly_fraction(0.60)
    0.0792

    .. _Kelly: https://en.wikipedia.org/wiki/Kelly_criterion
    .. _Thorp: https://www.jstor.org/stable/27641834
    """

    def __init__(self) -> None:
        """Initialise empty payoff history buckets.

        The history dictionaries are purposely *not* truncated.  If you intend
        to bound memory growth or decay the influence of obsolete data, hook
        your logic here.
        """
        self.p_bins = [0.5, 0.6, 0.7, 0.8, 0.9, 1.0]
        self.R_avg = {bin: [] for bin in self.p_bins}
        self.L_avg = {bin: [] for bin in self.p_bins}

    def update_historical_data(self, p: float, r: float) -> None:
        """Record a realised trade outcome and keep payoff buckets up‑to‑date.

        Parameters
        ----------
        p : float
            Model‑estimated probability of a *positive* return.  Must satisfy
            :math:`0 < p \\le 1`.
        r : float
            Realised fractional return, e.g. ``0.025`` represents **+2.5\xa0%**,
            ``-0.01`` represents **−1.0\xa0%**.

        Raises
        ------
        ValueError
            If *p* is outside the open interval *(0,\xa01]*.

        Notes
        -----
        The first confidence bin whose upper edge is ``>= p`` is chosen.  Gains
        (*r\xa0>\xa00*) are stored as‑is in :pyattr:`R_avg`; losses (*r\xa0<\xa00*) are
        stored as *their absolute value* in :pyattr:`L_avg` so that both lists
        contain positive magnitudes and can be averaged directly.
        """

    def get_R_L_for_p(self, p: float) -> tuple[float, float]:
        """Return the empirical mean gain/loss corresponding to *p*'s bin.

        Parameters
        ----------
        p : float
            Prediction confidence.

        Returns
        -------
        (float, float)
            ``(R_avg, L_avg)`` where *R_avg* is the mean positive return and
            *L_avg* is the mean positive loss seen in the same probability bin.
            If no data exist for a bin, a **safety floor** of *1\xa0basis‑point*
            (*0.0001*) is returned to avoid division‑by‑zero during Kelly
            optimisation.

        Warnings
        --------
        The returned statistics are *sample means*; the tails of the payoff
        distribution (higher moments) are ignored.  Use with caution if your
        strategy exhibits large kurtosis.
        """

    def calculate_kelly_fraction(self, p: float) -> float:
        """Compute the Kelly‑optimal capital fraction for the next trade.

        The method looks up :math:`R` and :math:`L` using
        :meth:`get_R_L_for_p`, then solves the maximisation problem numerically
        because a *closed‑form* does not exist for unequal payouts.  The search
        is bounded to the practical interval \\[0,\xa01\\].

        Parameters
        ----------
        p : float
            Model confidence.

        Returns
        -------
        float
            Optimal leverage fraction :math:`f^*` in the range \\[0,\xa01\\].  A
            value of ``0`` means *skip the trade*; a value of ``1`` means *go
            all‑in*.

        See Also
        --------
        get_R_L_for_p, update_historical_data
        """

def get_trailing_stop_loss(entry_price: float, current_price: float, max_price: float, trail_percent: float=0.02) -> tuple[float, float]:
    """Calculate a *moving* stop‑loss threshold that tightens as price rises.

    The stop level is expressed as a fixed percentage below the **maximum**
    price observed since entry, thereby *locking‑in* gains while giving the
    trade room to breathe.

    Parameters
    ----------
    entry_price : float
        Price at which the position was opened.  Included for convenience of
        end‑user taps but *not* used in the calculation.
    current_price : float
        Latest tick/quote.
    max_price : float
        Historical peak price already seen since the position was opened.
    trail_percent : float, optional
        Distance between *max_price* and the stop price, expressed as a decimal
        fraction.  Defaults to ``0.02`` (2\xa0%).

    Returns
    -------
    (float, float)
        ``(stop_loss, new_max_price)`` where *new_max_price* equals
        ``max(max_price, current_price)`` and *stop_loss* is simply
        ``new_max_price * (1 - trail_percent)``.

    Examples
    --------
    >>> stop, m = get_trailing_stop_loss(100, 106, 105, 0.03)
    >>> stop
    102.82

    Notes
    -----
    The function is *stateless*; you are responsible for persisting the updated
    *max_price* between calls (typically as a column in your positions table).
    """

def check_drawdown(current_balance: float, max_balance: float, threshold: float=0.2) -> bool:
    """Return *True* when a portfolio draw‑down crosses the danger threshold.

    Draw‑down is defined as the percentage drop from the historical equity
    peak::

        drawdown = (max_balance - current_balance) / max_balance

    A *True* signal can be routed directly into your scheduling logic to pause
    further order submission until manual review.

    Parameters
    ----------
    current_balance : float
        Present equity or cash value of the account.
    max_balance : float
        All‑time high equity achieved by the account.
    threshold : float, optional
        Trip‑wire level expressed as a decimal fraction.  A common choice is
        ``0.20`` for a 20\xa0% maximum peak‑to‑trough loss.

    Returns
    -------
    bool
        ``True`` if the computed draw‑down exceeds *threshold*, else ``False``.

    Warnings
    --------
    *max_balance* should be updated only *after* new equity highs are *fully
    realised*.  Mark‑to‑model valuations may lead to premature resets that
    desensitise the monitor.
