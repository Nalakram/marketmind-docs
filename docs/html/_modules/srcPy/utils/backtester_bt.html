class Backtester:

    def __init__(self, df, strategy_cls, strat_kwargs=None, cash=100000, commission=0.001, slippage_perc=0.0005, riskfreerate=0.02):
        """
        Initialize the Backtester with a DataFrame, strategy, and parameters.

        Args:
            df (pandas.DataFrame): Data with 'date' as index and OHLCV columns.
            strategy_cls (type): Backtrader strategy class.
            strat_kwargs (dict): Keyword arguments for the strategy.
            cash (float): Initial cash for the broker.
            commission (float): Commission rate per trade (e.g., 0.001 for 0.1%).
            slippage_perc (float): Slippage percentage (e.g., 0.0005 for 0.05%).
            riskfreerate (float): Annual risk-free rate (converted to daily internally).
        """
        self.df = df
        self.strategy_cls = strategy_cls
        self.strat_kwargs = strat_kwargs or {}
        self.cash = cash
        self.commission = commission
        self.slippage_perc = slippage_perc
        self.riskfreerate = riskfreerate / 252

    def run(self):
        """
        Run the backtest using Backtrader's Cerebro engine.

        Returns:
            dict: Performance metrics including Sharpe ratio, win rate, max drawdown, and daily returns.

        Raises:
            Exception: If backtest fails, logged and re-raised.
        """
        try:
            cerebro = bt.Cerebro()
            cerebro.broker.setcash(self.cash)
            cerebro.broker.setcommission(commission=self.commission)
            cerebro.broker.set_slippage_perc(self.slippage_perc)
            cerebro.broker.set_shortcash(True)
            data = bt.feeds.PandasData(dataname=self.df)
            cerebro.adddata(data)
            cerebro.addstrategy(self.strategy_cls, **self.strat_kwargs)
            cerebro.addanalyzer(bt.analyzers.SharpeRatio, _name='sharpe', riskfreerate=self.riskfreerate)
            cerebro.addanalyzer(bt.analyzers.DrawDown, _name='drawdown')
            cerebro.addanalyzer(bt.analyzers.TradeAnalyzer, _name='trades')
            results = cerebro.run(maxcpus=1)[0]
            sharpe = results.analyzers.sharpe.get_analysis().get('sharperatio', 0)
            drawdown = results.analyzers.drawdown.get_analysis().get('max', {}).get('drawdown', 0)
            trades = results.analyzers.trades.get_analysis()
            total_trades = trades.get('total', {}).get('total', 0)
            win_rate = trades.get('won', {}).get('total', 0) / total_trades if total_trades > 0 else 0
            closes = results.data.close.array[:len(results.data) - 1]
            daily_returns = pd.Series(closes).pct_change().dropna().values
            if len(daily_returns) < 1:
                raise ValueError('Insufficient data for daily returns calculation')
            return {'sharpe_ratio': sharpe, 'win_rate': win_rate, 'max_drawdown': drawdown, 'total_trades': total_trades, 'daily_returns': daily_returns}
        except Exception as e:
            logger.error(f'Backtest error: {e}')
            raise
