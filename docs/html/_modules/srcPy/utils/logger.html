"""MarketMind Logging Utilities
=============================

A **structured, multi‑sink** logging framework built on top of
:pyexternal+python:`logging` and :pymod:`structlog`. It is designed for the
observability needs of the *MarketMind* platform – from local notebooks to
Kubernetes‑hosted micro‑services.

Highlights
----------

* **Structured output** – Logs are enriched with key–value pairs and, when the
  handler formatter is set to ``json``, rendered as JSON suitable for searching
  in log aggregators (e.g. Loki, Elastic).
* **Pluggable sinks** – Supports console, rotating file, Syslog, HTTP webhooks,
  AWS CloudWatch, Google Cloud Logging and InfluxDB time‑series back‑ends.
* **Asynchronous I/O** – The queue/listener pattern keeps the hot path free of
  I/O latency by delegating heavy lifting to a background thread.
* **Sensitive‑data redaction** – A configurable list of keys (``api_key``,
  ``token`` …) is automatically masked before serialisation.
* **Dynamic configuration** – The single entry point :py:func:`configure_logger`
  accepts a self‑documenting *dict* so each deployment can tailor behaviour
  without touching code.

Quick start
~~~~~~~~~~~

>>> from marketmind.logger import configure_logger, get_logger
>>> configure_logger({"console": True, "file": True})
>>> log = get_logger(__name__)
>>> log.info("Order executed", order_id="A1B2C3", filled_qty=100)

Glossary
~~~~~~~~

``handler``
    A destination for log records (e.g. ``StreamHandler`` → *stdout*).

``processor``
    A :pyterm:`structlog` callable that mutates or augments the log *event_dict*
    before rendering (e.g. adds timestamps, masks secrets).

.. note::
   The defaults assume a directory called *logs/* at the project root; it is
   created automatically when missing.
"""

class InfluxDBHandler(logging.Handler):
    """Send log records to **InfluxDB** for time‑series analysis.

    Parameters
    ----------
    url
        The InfluxDB HTTP endpoint (including scheme and port).
    token
        Authentication token with permissions to write to *bucket*.
    org
        InfluxDB organisation slug.
    bucket
        Target bucket where points will be written.
    level
        Minimum severity level for records to reach this handler (defaults to
        :pydata:`logging.NOTSET`).
    """

    def __init__(self, url: str, token: str, org: str, bucket: str, level: int=logging.NOTSET):
        super().__init__(level)
        self.client = InfluxDBClient(url=url, token=token, org=org)
        self.write_api = self.client.write_api()
        self.bucket = bucket

    def emit(self, record: logging.LogRecord):
        """Translate *record* into an InfluxDB point and write it immediately."""
        point = {'measurement': record.name, 'tags': {'level': record.levelname}, 'fields': {'message': record.getMessage()}, 'time': record.created}
        try:
            self.write_api.write(bucket=self.bucket, record=point)
        except Exception:
            self.handleError(record)

def safe_filter_by_level(logger, method_name: str, event_dict: EventDict) -> EventDict:
    """Wrapper around :py:func:`structlog.stdlib.filter_by_level` resilient to *None*.

    ``structlog`` may call **filter_by_level** during initialisation when the
    *logger* instance is still ``None``. This helper guards against *Attribute
    Error* by short‑circuiting in that scenario.
    """
    if logger is None:
        return event_dict
    return original_filter_by_level(logger, method_name, event_dict)

def redact_sensitive_info(sensitive_keys: List[str]):
    """Return a processor that masks *sensitive_keys* inside *event_dict*.

    Any key present in *sensitive_keys* is replaced with the literal
    ``"***REDACTED***"`` **in‑place**. Example:

    >>> log.info("token refreshed", token="abc123")
    token=***REDACTED***

    Notes
    -----
    * The check is performed on the top‑level keys only; deeply nested secrets
      should be flattened by the caller before logging.
    """

    def processor(logger, log_method, event_dict):
        for key in sensitive_keys:
            if key in event_dict:
                event_dict[key] = '***REDACTED***'
        return event_dict
    return processor

def configure_logger(config: Optional[Dict[str, Any]]=None) -> None:
    """Initialise *structlog* and the standard logging system.

    The *config* dict merges into a verbose set of defaults (see source for the
    full schema). Missing keys therefore fall back to safe values. Handler
    level, formatter and enable‑flags are all controlled here.

    Parameters
    ----------
    config
        Optional mapping overriding the default settings. Refer to the *Quick
        start* section in the module docstring for the expected structure.

    Examples
    --------
    >>> configure_logger({
    ...     "console": True,
    ...     "file": True,
    ...     "root_level": logging.DEBUG,
    ...     "sensitive_keys": ["password", "token"],
    ... })

    After calling this function, use :py:func:`get_logger` to obtain module‑level
    loggers.
    """
    if config is None:
        config = {}
    default_config: Dict[str, Any] = {'root_level': logging.DEBUG, 'console': True, 'console_level': logging.INFO, 'file': True, 'file_level': logging.INFO, 'file_path': 'logs/marketmind.log', 'rotation': 'size', 'max_bytes': 10 * 1024 * 1024, 'backup_count': 5, 'when': 'midnight', 'interval': 1, 'async_mode': True, 'sensitive_keys': ['password', 'secret', 'api_key', 'token', 'auth', 'credentials'], 'handlers': {'console': 'console', 'file': 'json', 'syslog': 'json', 'http': 'json', 'cloudwatch': 'json', 'gcp': 'json', 'influxdb': 'json'}, 'syslog': {'enabled': False, 'address': ('localhost', 514), 'level': logging.INFO}, 'http': {'enabled': False, 'url': 'http://localhost:8000', 'method': 'POST', 'level': logging.ERROR}, 'cloudwatch': {'enabled': False, 'log_group': 'MarketMindLogs', 'level': logging.INFO}, 'gcp': {'enabled': False, 'level': logging.INFO}, 'influxdb': {'enabled': False, 'url': 'http://localhost:8086', 'token': '', 'org': '', 'bucket': 'marketmind_logs', 'level': logging.INFO}}
    config = {**default_config, **config}
    sensitive_keys = config.get('sensitive_keys', [])
    shared_processors = [merge_contextvars, safe_filter_by_level, structlog.stdlib.add_logger_name, structlog.stdlib.add_log_level, structlog.stdlib.PositionalArgumentsFormatter(), structlog.processors.TimeStamper(fmt='%Y-%m-%d %H:%M:%S', utc=False), redact_sensitive_info(sensitive_keys), structlog.processors.StackInfoRenderer(), structlog.processors.format_exc_info]
    formatters = {'console': structlog.stdlib.ProcessorFormatter(processor=structlog.dev.ConsoleRenderer(colors=True), foreign_pre_chain=shared_processors), 'json': structlog.stdlib.ProcessorFormatter(processor=structlog.processors.JSONRenderer(), foreign_pre_chain=shared_processors), 'plain': logging.Formatter('%(asctime)s − %(name)s − %(levelname)s − %(message)s')}

    def get_formatter(handler_name: str):
        formatter_type = config['handlers'].get(handler_name, 'json')
        return formatters.get(formatter_type, formatters['json'])
    handlers: List[logging.Handler] = []
    if config.get('console', False):
        console_handler = logging.StreamHandler(sys.stdout)
        console_handler.setLevel(config.get('console_level', logging.INFO))
        console_handler.setFormatter(get_formatter('console'))
        handlers.append(console_handler)
    if config.get('file', False):
        if config['rotation'] == 'time':
            file_handler = TimedRotatingFileHandler(config['file_path'], when=config['when'], interval=config['interval'], backupCount=config['backup_count'])
        else:
            file_handler = RotatingFileHandler(config['file_path'], maxBytes=config['max_bytes'], backupCount=config['backup_count'])
        file_handler.setLevel(config.get('file_level', logging.INFO))
        file_handler.setFormatter(get_formatter('file'))
        handlers.append(file_handler)
    if config['syslog']['enabled']:
        sys_handler = SysLogHandler(address=config['syslog']['address'])
        sys_handler.setLevel(config['syslog']['level'])
        sys_handler.setFormatter(get_formatter('syslog'))
        handlers.append(sys_handler)
    if config['http']['enabled']:
        http_handler = HTTPHandler(config['http']['url'], '/logs', method=config['http']['method'])
        http_handler.setLevel(config['http']['level'])
        http_handler.setFormatter(get_formatter('http'))
        handlers.append(http_handler)
    if config['cloudwatch']['enabled'] and watchtower:
        cw_handler = watchtower.CloudWatchLogHandler(log_group=config['cloudwatch']['log_group'])
        cw_handler.setLevel(config['cloudwatch']['level'])
        cw_handler.setFormatter(get_formatter('cloudwatch'))
        handlers.append(cw_handler)
    if config['gcp']['enabled'] and google_cloud_logging:
        client = google_cloud_logging.Client()
        gcp_handler = client.get_default_handler()
        gcp_handler.setLevel(config['gcp']['level'])
        gcp_handler.setFormatter(get_formatter('gcp'))
        handlers.append(gcp_handler)
    if config['influxdb']['enabled'] and InfluxDBClient:
        influx_handler = InfluxDBHandler(url=config['influxdb']['url'], token=config['influxdb']['token'], org=config['influxdb']['org'], bucket=config['influxdb']['bucket'], level=config['influxdb']['level'])
        handlers.append(influx_handler)
    root_logger = logging.getLogger()
    root_logger.setLevel(config['root_level'])
    root_logger.handlers.clear()
    if config.get('file', True):
        log_dir = Path(__file__).resolve().parent.parent / 'logs'
        log_dir.mkdir(exist_ok=True)
        file_path = log_dir / 'marketmind.log'
        file_handler = RotatingFileHandler(file_path, maxBytes=config.get('max_bytes', 10 * 1024 * 1024), backupCount=config.get('backup_count', 5))
        file_handler.setLevel(config.get('file_level', logging.DEBUG))
        root_logger.addHandler(file_handler)
    if config['async_mode']:
        log_queue: 'queue.Queue[logging.LogRecord]' = queue.Queue(-1)
        queue_handler = QueueHandler(log_queue)
        root_logger.addHandler(queue_handler)
        listener = QueueListener(log_queue, *handlers, respect_handler_level=True)
        listener.start()

        def stop_logging():
            """Flush and stop the background listener (registered on *atexit*)."""
            listener.stop()
        root_logger.stop_logging = stop_logging
    else:
        for h in handlers:
            root_logger.addHandler(h)
    structlog.configure(processors=[structlog.contextvars.merge_contextvars, safe_filter_by_level, structlog.stdlib.add_logger_name, structlog.stdlib.add_log_level, structlog.processors.TimeStamper(fmt='iso'), structlog.processors.format_exc_info, structlog.dev.ConsoleRenderer() if DEBUG else structlog.processors.JSONRenderer()], context_class=dict, logger_factory=structlog.stdlib.LoggerFactory(), wrapper_class=structlog.make_filtering_bound_logger(logging.DEBUG), cache_logger_on_first_use=True)

def get_logger(name: Optional[str]=None) -> BoundLogger:
    """Return a :class:`structlog.stdlib.BoundLogger` for *name*.

    If *name* is ``None`` the *root* logger is returned. Use this helper inside
    each module instead of calling :py:func:`structlog.get_logger` directly so
    the type checker can infer the correct return type.

    Parameters
    ----------
    name
        The logger namespace – usually ``__name__``. When omitted, a generic
        *MarketMind* logger is produced.

    Returns
    -------
    structlog.stdlib.BoundLogger
        A bound logger honoring the configuration established via
        :py:func:`configure_logger`.
    """
    if name:
        return structlog.get_logger(name)
    return structlog.get_logger()
