class AlphaVantageSource(DataSource):
    """Fetches market data from Alpha Vantage API."""

    def __init__(self, config):
        self.api_key = config.security.credentials.alpha_vantage_api_key
        if not self.api_key:
            raise ValueError('Alpha Vantage API key is required')
        self.base_url = 'https://www.alphavantage.co/query'



[docs]
    def get_historical(self, symbol: str, start: str, end: str) -> pd.DataFrame:
        params = {'function': 'TIME_SERIES_DAILY_ADJUSTED', 'symbol': symbol, 'apikey': self.api_key, 'outputsize': 'full'}
        try:
            with FETCH_DURATION.labels(source='AlphaVantage').time():
                response = requests.get(self.base_url, params=params)
                response.raise_for_status()
                data = response.json()
                if 'Time Series (Daily)' not in data:
                    raise DataFetchError(f'No historical data found for {symbol}')
                series = data['Time Series (Daily)']
                df = pd.DataFrame.from_dict(series, orient='index')
                df.index = pd.to_datetime(df.index)
                df = df.sort_index()
                df = df.loc[start:end]
                df = df.astype(float)
                df.columns = [col.split('. ')[1].lower() for col in df.columns]
            FETCH_SUCCESSES.labels(source='AlphaVantage').inc()
            return df
        except Exception as e:
            FETCH_FAILURES.labels(source='AlphaVantage').inc()
            logger.error('AlphaVantage historical fetch failed', extra={'symbol': symbol, 'error': str(e)})
            raise DataFetchError(f'Failed to fetch historical data for {symbol}', details={'error': str(e)})

    async def get_realtime(self, symbol: str) -> float:
        params = {'function': 'TIME_SERIES_INTRADAY', 'symbol': symbol, 'interval': '1min', 'apikey': self.api_key}
        try:
            async with aiohttp.ClientSession() as session:
                async with session.get(self.base_url, params=params) as response:
                    response.raise_for_status()
                    data = await response.json()
                    if 'Time Series (1min)' not in data:
                        raise DataFetchError(f'No intraday data found for {symbol}')
                    series = data['Time Series (1min)']
                    latest_timestamp = max(series.keys())
                    latest_data = series[latest_timestamp]
                    price = float(latest_data['4. close'])
            return price
        except Exception as e:
            FETCH_FAILURES.labels(source='AlphaVantage').inc()
            logger.error('AlphaVantage real-time fetch failed', extra={'symbol': symbol, 'error': str(e)})
            raise DataFetchError(f'Failed to fetch real-time data for {symbol}', details={'error': str(e)})

class CoinGeckoSource(DataSource):
    """Fetches crypto data from CoinGecko using pycoingecko."""

    def __init__(self):
        try:
            from pycoingecko import CoinGeckoAPI
            self.cg = CoinGeckoAPI()
        except ImportError:
            raise ImportError('pycoingecko is required for CoinGeckoSource. Please install it.')
        self.coin_map = {'BTC': 'bitcoin', 'ETH': 'ethereum', 'USDT': 'tether'}



[docs]
    def get_historical(self, symbol: str, start: str, end: str) -> pd.DataFrame:
        try:
            coin_id = self.coin_map.get(symbol.upper())
            if not coin_id:
                raise ValueError(f'Symbol {symbol} not supported by CoinGeckoSource.')
            from_ts = int(pd.Timestamp(start).timestamp())
            to_ts = int(pd.Timestamp(end).timestamp())
            with FETCH_DURATION.labels(source='CoinGecko').time():
                data = self.cg.get_coin_market_chart_range_by_id(coin_id, vs_currency='usd', from_timestamp=from_ts, to_timestamp=to_ts)
            prices = data['prices']
            df = pd.DataFrame(prices, columns=['timestamp', 'price'])
            df['Date'] = pd.to_datetime(df['timestamp'], unit='ms')
            df.set_index('Date', inplace=True)
            FETCH_SUCCESSES.labels(source='CoinGecko').inc()
            return df[['price']]
        except Exception as e:
            FETCH_FAILURES.labels(source='CoinGecko').inc()
            logger.error('CoinGecko fetch failed', extra={'symbol': symbol, 'error': str(e)})
            raise

class FileSource(DataSource):
    """Reads historical data from CSV or Parquet files."""

    def __init__(self, file_path: str):
        self.file_path = file_path
        self.ext = os.path.splitext(file_path)[1].lower()



[docs]
    def get_historical(self, symbol: str, start: str, end: str) -> pd.DataFrame:
        try:
            if self.ext == '.csv':
                df = pd.read_csv(self.file_path, parse_dates=True, index_col=0)
            elif self.ext == '.parquet':
                df = pd.read_parquet(self.file_path)
            else:
                raise ValueError(f'Unsupported file extension: {self.ext}')
            df = df.loc[(df.index >= start) & (df.index <= end)]
            return df
        except Exception as e:
            logger.error('FileSource fetch failed', extra={'file': self.file_path, 'error': str(e)})
            raise

def validate_dataframe(df: pd.DataFrame, required_columns: Optional[List[str]]=None) -> bool:
    if required_columns:
        missing_cols = [c for c in required_columns if c not in df.columns]
        if missing_cols:
            logger.error('DataFrame missing required columns', extra={'columns': missing_cols})
            raise ValueError(f'Missing columns: {missing_cols}')
    if df.index.duplicated().any():
        logger.error('Duplicate indices found in DataFrame')
        raise ValueError('Duplicate indices found')
    if df.isnull().any().any():
        logger.warning('NaNs detected in DataFrame')
    return True

def normalize(df: pd.DataFrame, columns: List[str]) -> pd.DataFrame:
    df = df.copy()
    for col in columns:
        min_val, max_val = (df[col].min(), df[col].max())
        df[col] = (df[col] - min_val) / (max_val - min_val + 1e-09)
    return df

def add_moving_average(df: pd.DataFrame, window: int=14, price_col: str='Close') -> pd.DataFrame:
    ma_col = f'MA_{window}'
    df[ma_col] = df[price_col].rolling(window).mean()
    return df

def add_rsi(df: pd.DataFrame, window: int=14, price_col: str='Close') -> pd.DataFrame:
    delta = df[price_col].diff()
    gain = delta.where(delta > 0, 0).rolling(window=window).mean()
    loss = -delta.where(delta < 0, 0).rolling(window=window).mean()
    rs = gain / (loss + 1e-09)
    df[f'RSI_{window}'] = 100 - 100 / (1 + rs)
    return df

class MarketDataManager:

    def __init__(self, config):
        self.config = config
        self.sources = []
        for source_cfg in config.market_data_sources:
            source_type = source_cfg['type']
            if source_type == 'alpha_vantage':
                self.sources.append(AlphaVantageSource(config))
            elif source_type == 'coingecko':
                self.sources.append(CoinGeckoSource())
            elif source_type == 'file':
                file_path = source_cfg.get('file_path')
                if not file_path:
                    raise ValueError('File path is required for file source')
                self.sources.append(FileSource(file_path))

    def get_historical(self, symbol: str, start: str, end: str, source_name: Optional[str]=None) -> pd.DataFrame:
        for src in self.sources:
            if source_name is None or src.__class__.__name__ == source_name:
                df = src.get_historical(symbol, start, end)
                validate_dataframe(df)
                return df
        raise ValueError(f'No data source found for: {source_name}')

        for src in self.sources:

    async def get_historical_batch(self, symbols: List[str], start: str, end: str, source_name: Optional[str]=None) -> Dict[str, pd.DataFrame]:
        loop = asyncio.get_event_loop()
        results = {}
        for symbol in symbols:
            df = await loop.run_in_executor(None, self.get_historical, symbol, start, end, source_name)
            results[symbol] = df
        return results
